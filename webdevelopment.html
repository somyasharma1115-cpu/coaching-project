🖥️ 3. Backend Development (Node.js + Express)

🔹 Node.js Basics

console.log("Hello Node");
const fs = require('fs');
fs.readFile('file.txt', (err, data) => { ... });
JavaScript runtime for server-side. Non-blocking, event-driven.

🔹 Express Server

const express = require('express');
const app = express();

app.get('/', (req, res) => {
res.send('Hello World');
});

app.listen(3000);

🔹 REST API & Middleware

app.use(express.json());

app.get('/api/users', (req, res) => {
res.json([{ id: 1, name: 'Alice' }]);
});

app.post('/api/users', (req, res) => {
const newUser = req.body;
// save to database
res.status(201).json(newUser);
});

🗄️ 4. Databases – SQL & NoSQL

🔹 SQL (PostgreSQL / MySQL)

CREATE TABLE users (id SERIAL PRIMARY KEY, name VARCHAR(100));
INSERT INTO users (name) VALUES ('Alice');
SELECT * FROM users WHERE id = 1;
Relational, ACID compliance. Use with Node.js via pg or mysql2.

🔹 MongoDB (NoSQL)

// Using Mongoose (ODM)
const userSchema = new mongoose.Schema({ name: String });
const User = mongoose.model('User', userSchema);
const user = new User({ name: 'Alice' });
await user.save();
Document-oriented, flexible schema, great for scaling.

🔹 Connecting Backend to DB

// MongoDB with Express
app.get('/api/users', async (req, res) => {
const users = await User.find();
res.json(users);
});

📁 7. Complete MERN Stack Example

🔹 Folder Structure & Setup

my-app/
├── client/ (React frontend)
├── server/ (Node + Express backend)
├── package.json

🔹 Backend (server/index.js)

const express = require('express');
const mongoose = require('mongoose');
const cors = require('cors');

const app = express();
app.use(cors());
app.use(express.json());

mongoose.connect('mongodb://localhost:27017/test');

const Item = mongoose.model('Item', { name: String });

app.get('/api/items', async (req, res) => {
res.json(await Item.find());
});

app.post('/api/items', async (req, res) => {
const newItem = new Item(req.body);
await newItem.save();
res.json(newItem);
});

app.listen(5000);

🔹 Frontend (client/App.js)

import { useState, useEffect } from 'react';
import axios from 'axios';

function App() {
const [items, setItems] = useState([]);
const [name, setName] = useState('');

useEffect(() => {
axios.get('http://localhost:5000/api/items')
.then(res => setItems(res.data));
}, []);

const addItem = async () => {
const res = await axios.post('http://localhost:5000/api/items', { name });
setItems([...items, res.data]);
setName('');
};

return (
<div>
<input value={name} onChange={e => setName(e.target.value)} />
<button onClick={addItem}>Add</button>
<ul>{items.map(i => <li key={i._id}>{i.name}</li>)}</ul>
</div>
);
}
