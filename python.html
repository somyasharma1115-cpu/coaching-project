⚙️ 2. Control Flow

🔹 if-elif-else

if score >= 90:
grade = "A"
elif score >= 80:
grade = "B"

🔹 Loops (for, while)

# for loop
for i in range(5): # 0..4

for item in [1,2,3]:
print(item)

# while loop
count = 0
while count < 3:
print(count)
count += 1

🔹 break, continue, else

if i == 5: break
if i % 2 == 0: continue
print(i)

🧩 3. Functions

🔹 Defining & Calling

def greet(name):
return f"Hello {name}"

🔹 Default & Keyword Arguments

def pow(base, exp=2):
return base ** exp

pow(5) # 25
pow(2,3) # 8
pow(exp=3, base=2) # keyword args

🔹 *args and **kwargs

def sum_all(*args):
return sum(args)

def print_info(**kwargs):
for k,v in kwargs.items():
print(f"{k}: {v}")

🔹 Lambda Functions

square = lambda x: x**2
nums = [1,2,3]
squared = list(map(lambda x: x**2, nums))

🧬 5. OOP in Python

🔹 Classes & Objects

class Car:
def __init__(self, brand):
self.brand = brand
def honk(self):
print(f"{self.brand} says Beep!")

my_car = Car("Toyota")
my_car.honk()

🔹 Inheritance

class Animal:
def speak(self): pass

class Dog(Animal):
def speak(self):
return "Woof!"

🔹 Encapsulation & Properties

class Person:
def __init__(self, age):
self._age = age # protected
@property
def age(self):
return self._age
@age.setter
def age(self, value):
if value > 0: self._age = value

🔹 Magic Methods (__str__, __repr__)

def __str__(self):
return f"Car({self.brand})"

🚀 10. Advanced Python

🔹 Decorators

def timer(func):
def wrapper(*args, **kwargs):
import time
start = time.time()
result = func(*args, **kwargs)
print(f"Time: {time.time()-start}")
return result
return wrapper

@timer
def slow_func():
import time; time.sleep(1)

🔹 Generators & yield

def count_up_to(n):
i = 0
while i < n:
yield i
i += 1

for num in count_up_to(5):
print(num)

🔹 List vs Generator Expressions

# List: memory heavy
squares = [x**2 for x in range(1000000)]
# Generator: lazy evaluation
squares_gen = (x**2 for x in range(1000000))

🔹 Context Managers (with statement)

class ManagedFile:
def __enter__(self):
self.file = open("file.txt", "w")
return self.file
def __exit__(self, exc_type, exc_val, exc_tb):
self.file.close()

🔹 Threading & AsyncIO

import threading
def worker(): print("Working")
t = threading.Thread(target=worker)
t.start()

# Async (asyncio)
import asyncio
async def main():
await asyncio.sleep(1)
print("Done")
