🟨 Complete JavaScript – The Language of the Web

JavaScript is the backbone of interactive web pages. This course covers everything: variables, functions, arrays, objects, DOM, events, async programming, ES6+, and more. Clear examples, no fluff. Let's become JS masters!

📖 1. JavaScript Fundamentals

🔹 Variables (var, let, const)

var old = "function-scoped";
let name = "block-scoped";
const PI = 3.14; // cannot reassign
Use let and const (ES6). var has function scope, avoid.

🔹 Data Types & typeof

let num = 42; // number
let str = "Hello"; // string
let bool = true; // boolean
let obj = {a:1}; // object
let arr = [1,2,3]; // object
let nothing = null; // object (quirk)
let notDefined; // undefined
console.log(typeof num);

🔹 Operators & Type Coercion

+ - * / % ++ --
> < >= <= == === != !==
&& || !
// == (value only), === (value + type)
console.log(5 == '5'); // true
console.log(5 === '5'); // false

🔹 Control Flow

if (age >= 18) {
console.log("Adult");
} else {
console.log("Minor");
}

for (let i=0; i<5; i++) { }
while (condition) { }
switch(day) { case 1: ... }

🧩 2. Functions

🔹 Function Declarations & Expressions

// Declaration (hoisted)
function add(a,b) { return a+b; }

// Expression
const multiply = function(a,b) { return a*b; };

🔹 Arrow Functions (ES6+)

const square = (x) => x * x;
const greet = name => `Hello ${name}`;
const sum = (a,b) => {
return a+b;
};
Shorter syntax, no own 'this' binding.

🔹 Parameters & Default Values

function greet(name = "Guest") {
console.log(`Hi ${name}`);
}
// Rest parameters
function sumAll(...nums) {
return nums.reduce((a,b)=>a+b,0);
}

🔹 Closures & Scope

function outer() {
let count = 0;
return function inner() {
count++;
return count;
};
}
const counter = outer();
counter(); // 1
Inner function remembers outer variables even after outer returns.

📊 3. Arrays & Objects

🔹 Arrays Methods

let arr = [1,2,3];
arr.push(4); // add to end
arr.pop(); // remove end
arr.unshift(0); // add to start
arr.shift(); // remove start
arr.map(x => x*2);
arr.filter(x => x>1);
arr.reduce((sum,x)=>sum+x,0);
arr.find(x => x===2);
arr.includes(3);

🔹 Object Literals & Methods

const person = {
name: "Alice",
age: 25,
greet() {
console.log(`Hi, I'm ${this.name}`);
}
};
person.greet();
// Dynamic keys
const key = "score";
person[key] = 100;

🔹 Destructuring & Spread/Rest

const [first, second] = [10,20];
const {name, age} = person;

const arr1 = [1,2];
const arr2 = [...arr1, 3,4]; // spread
function log(...args) { } // rest

🌳 4. DOM (Document Object Model)

🔹 Selecting Elements

document.getElementById('id');
document.querySelector('.class');
document.querySelectorAll('div');
document.getElementsByClassName('class');

🔹 Changing Content & Attributes

element.textContent = "New text";
element.innerHTML = "<strong>HTML</strong>";
element.setAttribute('src', 'image.jpg');
element.classList.add('active');
element.style.color = 'red';

🔹 Creating & Removing Elements

const div = document.createElement('div');
div.textContent = 'Hello';
parent.appendChild(div);
parent.removeChild(div);
// or element.remove()

🎯 5. Events

🔹 Adding Event Listeners

const btn = document.querySelector('button');
btn.addEventListener('click', () => {
alert('Clicked!');
});

// Common events: click, mouseover, keydown, submit, load, scroll

🔹 Event Object

btn.addEventListener('click', (e) => {
console.log(e.target); // element clicked
console.log(e.type);
e.preventDefault(); // stop default action
});

🔹 Event Delegation

document.querySelector('#parent').addEventListener('click', (e) => {
if (e.target.matches('.child')) {
console.log('Child clicked');
}
});

⏳ 6. Asynchronous Programming

🔹 Callbacks

function fetchData(callback) {
setTimeout(() => {
callback("Data loaded");
}, 1000);
}
fetchData(data => console.log(data));

🔹 Promises

const promise = new Promise((resolve, reject) => {
setTimeout(() => resolve("Success"), 1000);
});
promise.then(result => console.log(result))
.catch(err => console.error(err));

// Promise.all, Promise.race

🔹 Async/Await (ES8+)

async function getData() {
try {
const result = await fetch('https://api.example.com');
const data = await result.json();
console.log(data);
} catch(error) {
console.error(error);
}
}
Cleaner promise syntax. Always use try/catch.

🔹 Fetch API

fetch('https://jsonplaceholder.typicode.com/posts')
.then(res => res.json())
.then(data => console.log(data));

✨ 7. Modern JavaScript (ES6+)

🔹 Template Literals

const name = "John";
const msg = `Hello ${name},
welcome!`;

🔹 Classes

class Animal {
constructor(name) { this.name = name; }
speak() { console.log(`${this.name} makes noise`); }
}
class Dog extends Animal {
speak() { console.log(`${this.name} barks`); }
}

🔹 Modules (import/export)

// math.js
export const add = (a,b) => a+b;
// main.js
import { add } from './math.js';
Use type="module" in script tag.

🔹 Optional Chaining & Nullish Coalescing

const user = {};
console.log(user?.profile?.name); // undefined
const value = input ?? "default"; // if null/undefined

🔹 Map, Set, WeakMap, WeakSet

const map = new Map();
map.set('key', 'value');
const set = new Set([1,2,2,3]); // unique values

🌐 8. Browser APIs & Advanced

🔹 LocalStorage & SessionStorage

localStorage.setItem('key', 'value');
const val = localStorage.getItem('key');
sessionStorage.setItem('temp', 'data');
localStorage.removeItem('key');

🔹 Timers (setTimeout, setInterval)

setTimeout(() => console.log("After 1s"), 1000);
const interval = setInterval(() => console.log("tick"), 1000);
clearInterval(interval);

🔹 JSON

const obj = { name: "Alice" };
const jsonStr = JSON.stringify(obj);
const newObj = JSON.parse(jsonStr);

🔹 Error Handling (try/catch/finally)

try {
throw new Error("Oops");
} catch (err) {
console.log(err.message);
} finally {
console.log("Cleanup");
}

🔹 Regular Expressions

const pattern = /hello/i;
pattern.test("Hello world"); // true
"abc123".match(/\d+/); // ["123"]

🐞 9. Debugging & Best Practices

🔹 Console Methods

console.log, console.error, console.warn, console.table, console.time, console.group

🔹 Debugger & DevTools

Use 'debugger;' statement. Sources tab breakpoints.

🔹 Strict Mode

// At top of file/function
"use strict";
